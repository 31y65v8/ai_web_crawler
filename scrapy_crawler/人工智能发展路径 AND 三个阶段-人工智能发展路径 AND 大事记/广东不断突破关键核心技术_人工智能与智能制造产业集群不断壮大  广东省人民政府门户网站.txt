            var child = p.children[i];

                cb(child);
            }
            // 递归

            <p style="text-align: justify;">　　2016年3月，在举世瞩目的“人机大战”中，谷歌“阿尔法狗”以4∶1战胜围棋世界冠军李世石。有人评论：“机器战胜了人类，人工智能已经到达了取代人类的奇点。”</p><p style="text-align: justify;">　　当时，科学家韩旭观看了那场比赛，他敏锐地意识到，人工智能将成为各国博弈的重要焦点。2017年，韩旭正式开始在硅谷创业，并在当年底选择在广州创办文远知行。如今，这家企业已成为全球领先的自动驾驶公司。</p><p style="text-align: justify;">　　人工智能最早发端于美国，并迅速在中国得到应用，特别是在广东，打造了一个个丰富多彩的应用场景，而伴随着应用层的不断深入，广东在技术层、基础层不断突破关键核心技术，走出了一条从并跑到领跑的发展路径。</p><p style="text-align: justify;">　　作为新一轮产业变革的核心驱动力，人工智能与实体经济加速融合，在广东涌现一批新产业、新业态，不仅为传统企业带来了新的生机，还形成了人工智能与智能制造产业集群，成为推动经济发展的新动能、新引擎。</p><p style="text-align: justify;"><strong>　　向实体经济融合</strong></p><p style="text-align: justify;"><strong>　　智能决策助力各行业降本增效</strong></p><p style="text-align: justify;">　　走出广州官洲地铁站，约上一辆网约车，来到跟前的可能是一辆无人驾驶小巴，没有方向盘也没有后视镜，刷新了乘客对未来出行的认知。</p><p style="text-align: justify;">　　“广州的气候几乎满足了无人驾驶测试的所有条件。”在创业之初，韩旭和团队就发现，广州暴雨天气频发，普通人在强降水中也会视线模糊，传感器更是会被雨水干扰；广州江河、桥梁、隧道众多，对无人驾驶形成巨大挑战。</p><p style="text-align: justify;">　　经过多次技术讨论、设备调试、数据采集、代码验证、模拟测试，2018年6月，文远知行最终依靠感知、决策规划、定位等全方面人工智能算法的互相配合，在业内首次完成了暴风雨中无GPS信号情况下无人驾驶穿越江底隧道。如今，文远知行又在业内率先开展无人驾驶出租车商业化运营。</p><p style="text-align: justify;">　　在围棋“人机大战”后短短两年时间，人工智能的火种迅速在广东点燃。</p><p style="text-align: justify;">　　广东陆续涌现云从科技、云天励飞、文远知行、小马智行、安途智行等一批人工智能创业公司，广东丰富的应用场景吸引这些公司的创始人前来创业，各自在视觉检测、芯片、自动驾驶等领域走出差异化道路。</p><p style="text-align: justify;">　　根据2018年印发的《广东省新一代人工智能发展规划》，广东人工智能核心产业规模约260亿元，约占全国1/3，带动机器人及智能装备等相关产业规模超2000亿元，人工智能核心产业及相关产业规模均居全国前列。</p><p style="text-align: justify;">　　与此同时，广东本土人工智能研发力量迅速崛起，工业机器人、民用无人机、智能手机等智能产品不断推陈出新，出货量位居全国前列，智能制造、智能医疗、智能家居、智能农业等新产业、新业态不断涌现。人工智能正以一种“强势”的姿态，一头扎进产业互联网深水区，推动实体经济智能化升级。</p><p style="text-align: justify;">　　“5G和人工智能等技术的提升，使数据采集、数据治理更加高效，数字化传承取代了老师傅经验。”裕申科技副总经理翁子舜说。</p><p style="text-align: justify;">　　“最现代+最古老”的组合，更悄然改变着传统农业“靠天吃饭”的局面。在广西南宁市武鸣区江龙村的沃柑园，一眼望去，青涩小果掩映在葱郁的绿叶中，无人机腾空而起，喷洒出雾状药液。此前，果园植保以人工拉管打药方式完成，80亩果园、5000多棵果树，需要五六名工人连续作业2天，现在，两架大疆无人机4小时就能完成。</p><p style="text-align: justify;">　　“人工智能已深深嵌入不同场景的生产作业流程中，像电力和互联网一样，成为一类新的通用技术。”华为轮值董事长胡厚崑说。</p><p style="text-align: justify;"><strong>　　向技术平台跨越</strong></p><p style="text-align: justify;"><strong>　　赋能中小企业集体数字化</strong></p><p style="text-align: justify;">　　2020年，新冠肺炎疫情让人工智能应用按下了“快进键”，从健康码“一码通行”，到无人装备的投入，再到对新冠病毒的检测预防，大数据、人工智能、云计算等新技术轮番上场，推动政府科学决策、精细治理。</p><p style="text-align: justify;">　　疫情不仅推动人工智能单点技术突破，也在推动人工智能向系统性、综合性解决问题。“比如，医疗领域始终面临稀缺的专家资源与海量的病患需求之间的矛盾。我们就在思考，如何将医疗专家的知识技能模型化、自动化，让系统可以自动判断90%以上诊疗信息，医疗专家则集中处理10%的关键问题。”云从科技董事长周曦说。</p><p style="text-align: justify;">　　对此，云从科技推出了承载平台生态化的人机协同操作系统，它就像一个“AI超市”，集合了算法训练、智能调度、数据治理等功能模块，用户可以像组建积木一样，随时调用这些功能。</p><p style="text-align: justify;">　　技术平台的建立，使人工智能技术不仅可以自用，还可以更好地开放出来赋能传统中小企业。针对智能化面临的“不敢、不能”等转型痛点，平台纷纷下沉到传统企业集聚区，推动产业集群整体智能化跃迁。</p><p style="text-align: justify;">　　每当有新产品研发出来，东莞慕思股份都会更新在工业互联网标识解析体系中，避免信息不对称的发生，而供应商也借助这套创新体系实现数字化转型。慕思流程信息中心IT总监余启龙介绍，对上下游供应商来说，这套数字化智能化系统就像水龙头，拧开就能用，中小企业智能化转型的积极性高涨。</p><p style="text-align: justify;">　　“同一个行业的定制开发是相通的，这些经验可以沉淀到智能化平台上，中小企业就可以低成本拿来用。”树根互联联合创始人、CEO贺东东说。</p><p style="text-align: justify;">　　瞄准传统产业中小企业多、生产零散的实际，《广东省制造业高质量发展“十四五”规划》提出，推动20个战略性产业集群数字化、智能化转型。政企联手，为人工智能深入行业应用打下了坚实的基础。</p><p style="text-align: justify;"><strong>　　向核心技术挺进</strong></p><p style="text-align: justify;"><strong>　　产品创新能力持续增强</strong></p><p style="text-align: justify;">　　伴随着中国在人工智能领域从跟跑到领跑，在一些技术上进入“无人区”，尤其是国际环境风云变幻之下，广东已经意识到，发展离不开源头核心技术的创新。</p><p style="text-align: justify;">　　会跳舞、练瑜伽、能爬楼梯，还能端茶倒水……曾经在春晚舞台上一炮而红的人形机器人优必选，最近发布了智慧康养战略及全体系解决方案，一些机器人陆续走进医院、养老院，为用户提供智慧康养服务。</p><p style="text-align: justify;">　　这背后，是优必选对人工智能基础层核心技术持续不懈的攻关。智能机器人涉及几千个零部件，还要在软件设计、导航、语音、视觉等算法及伺服驱动器等软硬集成上反复琢磨，研发过程中最难的是“伺服驱动器”，也就是能让人形机器人自由活动的关节。单这一个关节，团队就钻研了五年。</p><p style="text-align: justify;">　　“在智能机器人这个领域，我们就像一只‘拓荒牛’一样，长期在无人区行进。”优必选科技创始人、董事长兼CEO周剑说。</p><p style="text-align: justify;">　　而作为一家配送机器人公司，赛特智能非常重视定位、导航等核心技术。公司董事长李睿注意到，疫情期间多地为减少接触而引入配送机器人，但不同机器人对环境的适应能力千差万别，有些地方甚至还要抽人去照看机器人，“只有突破核心技术，才能谈实际落地应用”。</p><p style="text-align: justify;">　　一直以来，我国人工智能产业侧重于技术层和应用层，终端产品尤其丰富，但受限于创新难度大、技术和资金壁垒高等问题，基础层短板突出，底层技术和基础理论缺乏标志性研究成果。以优必选、赛特智能为代表，越来越多的AI技术创新向最难啃的“硬骨头”发起冲击。</p><p style="text-align: justify;">　　广东还在积极谋划新一代人工智能产业的系统性战略布局，部分领域关键核心技术取得重大突破，一批具有地域特色的开放创新平台成为行业标杆，同时，围绕创新链布局产业链，广东已建设广州人工智能公共算力中心、国家超级计算广州中心等一批创新平台。目前，广东催生了一批人工智能核心产业，智能芯片、开源框架等关键核心技术取得重要突破，智能芯片、终端、机器人等标志性产品的创新能力持续增强。</p><p style="text-align: justify;">　　广东将继续培育建设一批人工智能产业园区，到2025年，产业核心规模突破1500亿元，带动相关产业规模达到1.8万亿元；到2030年，人工智能产业发展进入全球价值链高端环节。</p><p style="text-align: justify;">　　“下一步，人工智能产业发展的关键在于规划好创新方向，真正让人工智能为各行各业的数字化升级创造价值。”胡厚崑说。</p><p style="text-align: justify;"><strong>　　新经济时评</strong></p><p style="text-align: justify;"><strong>　　多个“国内首部”条例</strong></p><p style="text-align: justify;"><strong>　　为广东领跑人工智能提供坚实保障</strong></p><p style="text-align: justify;">　　继蒸汽、电力、计算机为代表的前三次工业革命以后，以人工智能为代表的第四次工业革命已然到来。不同于以往生产方式的变革，中国及中国企业只被动接受了工业发展的成果，在第四次工业革命中，中国可以更主动参与其中。</p><p style="text-align: justify;">　　在发展人工智能的路径选择上，发达国家在原理性研究、理论性研究上有先发优势。而中国则依托拥有广阔的市场、海量的数据，从场景、应用着手反哺底层技术发展，不断补上短板，走出了一条适合本地发展的路径。</p><p style="text-align: justify;">　　值得注意的是，伴随着中国人工智能产业发展不断在“无人区”领跑，“拿来主义”已经行不通，必然要强化原始创新、底层创新，才能在这种高科技的竞逐中持续领先。为人工智能发展保驾护航，近年来广东围绕人工智能立法动作不断——</p><p style="text-align: justify;">　　2022年9月6日，《深圳经济特区人工智能产业促进条例》正式公布。该条例是我国首部人工智能产业专项立法，拟于今年11月1日起实施。条例从深圳人工智能产业发展实际出发，围绕“明确范围+补齐短板+强化支撑+抢抓应用+集聚发展+规范治理”等环节进行探索创新。</p><p style="text-align: justify;">　　8月，国内首部关于智能网联汽车管理的法规《深圳经济特区智能网联汽车管理条例》正式实施，无人驾驶汽车可以合法上路，这为人工智能挺进提供了新机遇。</p><p style="text-align: justify;">　　5月，国内首部城市数字经济地方性法规《广州市数字经济促进条例》发布，并于6月1日起实施。</p><p style="text-align: justify;">　　因为规模优势初具，且占据举足轻重的地位，广东多个城市不断探索并最终出台多部产业专项立法，可谓是情理之中。据天眼查数据，目前我国已有104.4万家数字经济相关企业，广东5G、人工智能相关企业数量均居全国第一。2021年全年，深圳数字经济核心产业增加值占GDP比重约30%，规模和质量均居全国大中城市前列。</p><p style="text-align: justify;">　　与此同时，人工智能技术目前正处于科技发展的前沿，需要用法治建设保障创新环境。坚持改革决策和立法决策相衔接，加快重要领域和新兴领域立法，唯有此，人们才能更快地看到其真正实现市场转化的曙光。</p><p style="text-align: justify;">　　事实上，新生事物并非排斥法律法规的制约，相反，法律对于新兴商业模式和技术创新的有效规范和制约能更好地引导企业、行业健康有序发展。</p><p style="text-align: justify;">　　例如，为破解人工智能产品落地难问题，深圳就提出创新产品准入制度，对于国家、地方尚未制定标准但符合国际先进产品标准或者规范的低风险人工智能产品和服务，允许通过测试、试验、试点等方式开展先行先试。</p><p style="text-align: justify;">　　以地方立法推动人工智能发展，解决产业发展中存在的种种难点问题，有助于贯彻落实国家战略部署，加快发展新一代人工智能，为广东建设具有全球影响力的科技创新中心提供坚实的法治保障。加快数字经济与人工智能立法，越早行动，就越主动。</p><p style="text-align: justify;"><strong>　　广东人工智能发展大事记</strong></p><p style="text-align: justify;"><strong>　　2017年</strong></p><p style="text-align: justify;">　　●广州确立了IAB(新一代信息技术、人工智能、生物医药)产业发展战略。</p><p style="text-align: justify;">　　●华为在柏林公布最新的麒麟970芯片，这是世界首款带了专用人工智能元素的手机芯片。</p><p style="text-align: justify;"><strong>　　2018年</strong></p><p style="text-align: justify;">　　●广东发布《广东省新一代人工智能发展规划》并提出，到2025年产业核心规模突破1500亿元，带动相关产业规模达到1.8万亿元。</p><p style="text-align: justify;"><strong>　　2021年</strong></p><p style="text-align: justify;">　　●《广州市人工智能产业链高质量发展三年行动计划（2021—2023年）》发布。</p><p style="text-align: justify;"><strong>　　2022年</strong></p><p style="text-align: justify;">　　●国内首部城市数字经济地方性法规——《广州市数字经济促进条例》发布。</p><p style="text-align: justify;">　　●科创板“AI平台第一股”云从科技上市。</p><p style="text-align: justify;">　　●国内首部关于智能网联汽车管理的法规《深圳经济特区智能网联汽车管理条例》正式实施。</p><p style="text-align: justify;">　　●国内首部人工智能立法《深圳经济特区人工智能产业促进条例》正式公布。</p>

function Traditionalized(cc) {
	var str = '', ss = JTPYStr(), tt = FTPYStr();
	for (var i = 0; i < cc.length; i++) {
		if (cc.charCodeAt(i) > 10000 && ss.indexOf(cc.charAt(i)) != -1) str += tt.charAt(ss.indexOf(cc.charAt(i)));
		else str += cc.charAt(i);
	}
	return str;
}

function setCookie(name, value) {
	var argv = setCookie.arguments;
	var argc = setCookie.arguments.length;
	var expires = (argc > 2) ? argv[2] : null;
	if (expires != null) {
		var LargeExpDate = new Date();
		LargeExpDate.setTime(LargeExpDate.getTime() + (expires * 1000 * 3600 * 24));
	}
	document.cookie = name + "=" + escape(value) + ((expires == null) ? "" : ("; expires=" + LargeExpDate.toGMTString()));
}

function getCookie(Name) {
	var search = Name + "="
	if (document.cookie.length > 0) {
		offset = document.cookie.indexOf(search)
		if (offset != -1) {
			offset += search.length
			end = document.cookie.indexOf(";", offset)
			if (end == -1) end = document.cookie.length
			return unescape(document.cookie.substring(offset, end))
		}
		else return ""
	}
}
